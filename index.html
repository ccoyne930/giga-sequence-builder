  const tags = [];

  const tagInput = document.getElementById('tagInput');

  tagInput.addEventListener('keydown', e => {

    if ((e.key === 'Enter' || e.key === ',') && tagInput.value.trim()) {

      e.preventDefault();

      addTag(tagInput.value.trim());

      tagInput.value = '';

    } else if (e.key === 'Backspace' && !tagInput.value && tags.length) {

      removeTag(tags.length - 1);

    }

  });

  function addTag(val) { tags.push(val); renderTags(); }

  function removeTag(i) { tags.splice(i, 1); renderTags(); }

  function renderTags() {

    document.querySelectorAll('.tag').forEach(t => t.remove());

    const wrap = document.getElementById('tagWrap');

    tags.forEach((t, i) => {

      const el = document.createElement('div');

      el.className = 'tag';

      el.innerHTML = `${t}<button type="button" onclick="removeTag(${i})">×</button>`;

      wrap.insertBefore(el, tagInput);

    });

  }
